    SQL INSERT, DELELE, UPDATE, MERGE TABLE

    Nov 11, 2021

    SELECT

    • SELECT is statement read data from specific table.

    • Table PERSONS

    refPerson preName name doB legalName
    01A100 WL. Morman 1990  
    02B101 W. Fred 1980  
    03B103 H. Han 1992

    05V104 G. Alan 1875 Agency CL - L1
    • Ex: Get name and date of birth
    SELECT
  name
, doB
FROM  PERSONS;
    • Ex: Get all info in table PERSONS
    SELECT * FROM  PERSONS;
    • SELECT statement compile with WHERE conditions that filter data. I will have other topic go to WHERE condition intensive.

    INSERT new record into table.

    • Note: number of column must equal number of value. If have no value, leave it NULL
    • Column name must not duplicate
    • Syntax 
      INSERT INTO tableName (column1 , column2, .. , columnN)
VALUES ('value1', 'value2', .. , 'valueN');
    • Example
    INSERT INTO PERSONS (preName , name, doB, legalName)
VALUES ('JU.', 'LUC', '2020-12-29', NULL);

    DELETE record from table.

    • Syntax 
      DELETE  tableName
WHERE expression = condition;
    • Example: Delete row in table PERSON which has refPerson equal 01A100
    DELETE  PERSONS
WHERE  refPerson = '01A100';

    UPDATE record in table.

    • Syntax 
      UPDATE  tableName
SET column1 = value1
, columnN = valueN
WHERE expression = condition;
    • Example: Update person’s preName / name to ‘MRUPDATE’ whose refPerson equal 01A100
    UPDATE  PERSONS
SET preName = 'M.'
, name = 'DENIS'
WHERE  refPerson = '01A100';

    MERGE.

    • The MERGE statement selects the rows from one or more tables (called Source table), and based on conditions specified, INSERT or UPDATE data to another table (called Target table). 
      MERGE [Destination table] AS ds
USING  [Source Table] AS st
ON ds.[Common column] = st.[Common Column]
WHEN MATCHED THEN INSERT | DELETE | UPDATE
WHEN NOT MATCHED THEN INSERT | DELETE | UPDATE

    • Update record when it existed in table. If not, create new one. You can do above by code.

    //PersonDao.java

@Repository
public class PersonDao {
  private final NamedParameterJdbcOperations namedTemplate;

public Optional<Person> getPersonById(final String refPerson) {

    String sql = " SELECT refPerson, preName , name, doB, legalName "
               + "  FROM PERSONS                                    "
               + "  WHERE refPerson  = :refPerson                   "
    ;
    MapSqlParameterSource map = new MapSqlParameterSource()
      .addValue("refPerson",  refPerson);

    try {
      return Optional.ofNullable(namedTemplate.queryForObject(sql, map, (rs, r) -> {
        final Person p = new Person();
        p.setRefPerson( rs.getString("refPerson"));
        p.setPreName(rs.getString("preName"));
        p.setName(rs.getString("name"));
        p.setLegalName(rs.getString("legalName"));
        p.setDateOfBirth(rs.getObject("doB", LocalDate.class));
        return p;
      })
      );
    } catch (EmptyResultDataAccessException e) {
      return Optional.empty();
    }
  }

public void  createPerson (PersonAdd toCreate){
  String sql =
    " INSERT INTO PERSONS (preName , name, doB, legalName) "
  + " VALUES (:preName, :name, :dob, :legalName)           "
  MapSqlParameterSource map = new MapSqlParameterSource()
    .addValue("preName",  toCreate.getPreName());
    .addValue("name",  toCreate.getName());
    .addValue("doB",  toCreate.setDateOfBirth());
    .addValue("legalName",  toCreate.getLegalName());
    namedTemplate.update(sql, map);


  public void  updatePerson (PersonUpdate toUpdate){
      String sql =
         " UPDATE PERSONS SET                "
       + " SET preName = :preName            "
       + " , name = :name                    "
       + " , doB = :doB                      "
       + " , legalName = :legalName          "
       + " WHERE refPerson  = :refPerson     ";

      MapSqlParameterSource map = new MapSqlParameterSource()
        .addValue("preName",  toUpdate.getPreName());
        .addValue("name",  toUpdate.getName());
        .addValue("doB",  toUpdate.setDateOfBirth());
        .addValue("legalName",  toUpdate.getLegalName());
        namedTemplate.update(sql, map);    

}

    • When compile MERGE with Java, this statement make your code more shorter

  personDao.mergePerson("01A100", person);

    //PersonDao.java
@Repository
public class PersonDao {
  private final NamedParameterJdbcOperations namedTemplate;

  public void  mergePerson (PersonUpdate toMerge){
      String sql =
    " MERGE INTO PERSONS P                 "                                     
  + " USING (SELECT  :refPerson ,          "
  + "                :preName   ,          "
  + "                :name      ,          "
  + "                :doB       ,          "
  + "                :legalName            "
  + "           FROM dual) SRC             "                        
  + "    ON (P.refPerson = SRC.refPerson)  "                             
  + " WHEN MATCHED THEN UPDATE             "                            
  + "    SET P.preName   = SRC.preName  ,  "         
  + "        P.name      = SRC.name     ,  "         
  + "        P.doB       = SRC.doB      ,  "         
  + "        P.legalName = SRC.legalName   "                
  + " WHEN NOT MATCHED THEN                "                            
  + "   INSERT                             "                            
  + "   (refPerson,                        "               
  + "    preName  ,                        "               
  + "    name     ,                        "               
  + "    doB      ,                        "               
  + "    legalName)                        "                       
  + "   VALUES (SRC.refPerson,             "                      
  + "           SRC.preName  ,             "                      
  + "           SRC.name     ,             "                      
  + "           SRC.doB      ,             "                      
  + "           SRC.legalName )            ";
      MapSqlParameterSource map = new MapSqlParameterSource()
        .addValue("RefPerson",  toMerge.getRefPerson());
        .addValue("preName",  toMerge.getPreName());
        .addValue("name",  toMerge.getName());
        .addValue("doB",  toMerge.setDateOfBirth());
        .addValue("legalName",  toMerge.getLegalName());
        namedTemplate.update(sql, map);    
}
