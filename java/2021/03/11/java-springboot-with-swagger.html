    WHAT IS THE SWAGGER?

    • Swagger allows you to describe the structure of your APIs so that machines can read them. The ability of APIs to describe their own structure is the root of all awesomeness in Swagger. Why is it so great? Well, by reading your APIâ€™s structure, we can automatically build beautiful and interactive API documentation. We can also automatically generate client libraries for your API in many languages and explore other possibilities like automated testing. Swagger does this by asking your API to return a YAML or JSON that contains a detailed description of your entire API. This file is essentially a resource listing of your API which adheres to OpenAPI Specification.

    DEPENDENCIES AND PLUGINS

    • You need to import some dependencies and plugins below into pom.xml of your spring boot project

        ....

		<springfox.version>2.9.2</springfox.version>
        ....
	</properties>

    <!-- swagger -->

        <artifactId>swagger-codegen</artifactId>
        <version>${swagger.version}</version>
    </dependency>
    <dependency>
        <groupId>io.springfox</groupId>
        <artifactId>springfox-swagger2</artifactId>

    </dependency>

    </dependency>
    <dependency>
        <groupId>com.fasterxml.jackson.datatype</groupId>
        <artifactId>jackson-datatype-joda</artifactId>
    </dependency>


	<plugins>
    ...
        <plugin>
            <groupId>org.codehaus.mojo</groupId>
            <artifactId>build-helper-maven-plugin</artifactId>
            <version>1.9</version>
            <executions>
                <execution>
                    <id>add-source</id>
                    <phase>generate-sources</phase>
                    <goals>
                        <goal>add-source</goal>
                    </goals>
                    <configuration>
                        <sources>
                            <source>${basedir}/target/generated-sources/src/main/java</source>
                        </sources>
                    </configuration>
                </execution>
            </executions>
        </plugin>
        <plugin>
            <groupId>io.swagger</groupId>
            <artifactId>swagger-codegen-maven-plugin</artifactId>
            <version>2.3.1</version>
            <executions>
                <execution>
                    <id>swagger-code-generate</id>
                    <goals>
                        <goal>generate</goal>
                    </goals>
                    <configuration>
                        <!-- path to the swagger file `.yml` -->
                        <inputSpec>${project.basedir}/src/main/resources/swagger/swagger-input.yml
                        </inputSpec>
                        <language>spring</language>
                        <!-- path to the configuration file -->
                        <configurationFile>
                            ${project.basedir}/src/main/resources/swagger/service-config.json
                        </configurationFile>
                        <!-- output directory, place contains generated classes -->
                        <output>${project.build.directory}</output>
                        <!-- path to the resource package that contain configuration templates -->
                        <templateDirectory>src/main/resources/swagger/templates</templateDirectory>
                    </configuration>
                </execution>
            </executions>
        </plugin>
	</plugins>

    TESTING

    • Now, you will use command mvn clean package to build your project with swagger then you will see the generated source as in the image below

    • Now, let create a DelegateImpl component and implement V1ApiDelegate from generated source, then add business code as the example below
    
    package com.example.todos.delegate;

    import com.example.todos.api.V1ApiDelegate;
    import com.example.todos.api.model.*;
    import com.example.todos.services.AuthenticationService;
    import com.example.todos.services.TodoService;
    import com.example.todos.services.UserService;
    import org.springframework.beans.factory.annotation.Autowired;
    import org.springframework.http.HttpStatus;
    import org.springframework.http.ResponseEntity;
    import org.springframework.stereotype.Component;

    import java.util.List;

    @Component
    public class V1ApiDelegateImpl implements V1ApiDelegate{

        @Autowired
        private TodoService todoService;

        @Autowired
        private UserService userService;

        @Autowired
        private AuthenticationService authenticationService;

        @Override
        public ResponseEntity<Todo> createTodo(String username, Todo body) {
            return new ResponseEntity<>(todoService.createTodo(username, body), HttpStatus.OK);
        }

        @Override
        public ResponseEntity<String> createUser(String username, String password) {
            return ResponseEntity.ok(userService.createUser(username, password));
        }

        @Override
        public ResponseEntity<Boolean> deleteTodoById(String username, String id) {
            return new ResponseEntity<>(todoService.deleteTodo(username, id), HttpStatus.OK);
        }

        @Override
        public ResponseEntity<Todo> getTodoById(String username, String id) {
            return new ResponseEntity<>(todoService.getTodoById(username, id), HttpStatus.OK);
        }

        @Override
        public ResponseEntity<List<Todo>> getTodosByUsername(String username) {
            return new ResponseEntity<>(todoService.getTodosByUsername(username), HttpStatus.OK);
        }

        @Override
        public ResponseEntity<JwtTokenResponse> getToken(JwtTokenRequest authenticationRequest) {
            return new ResponseEntity<>(authenticationService.getToken(authenticationRequest), HttpStatus.OK);
        }

        @Override
        public ResponseEntity<JwtTokenResponse> refreshToken(String requestHeader) {
            JwtTokenResponse token = authenticationService.refreshToken(requestHeader);
            if(token != null) {
                return new ResponseEntity<>(token, HttpStatus.OK);
            }
            throw new RuntimeException("Token can not be refeshed!");
        }

        @Override
        public ResponseEntity<Boolean> saveTodoById(String username, String id, Todo todo) {
            return new ResponseEntity<>(todoService.saveTodoById(username, id, todo), HttpStatus.OK);
        }

        @Override
        public ResponseEntity<Welcome> welcomeWithUsername(String name) {
            Welcome welcome = new Welcome();
            welcome.setMessage("Hi " + name + ", Welcome to my Todo site");
            return new ResponseEntity<>(welcome, HttpStatus.OK);
        }

        @Override
        public ResponseEntity<Welcome> welcome() {
            Welcome welcome = new Welcome();
            welcome.setMessage("Welcome to my Todo site");
            return new ResponseEntity<>(welcome, HttpStatus.OK);
        }
    }
