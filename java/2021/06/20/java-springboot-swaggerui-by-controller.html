    Generate Swagger UI from Controller in Spring Boot

    Jun 20, 2021

    In this topic we will learn how to generate swagger ui and openapi ui from Controller class in our spring boot project.

    • GENERATE SWAGGER UI FROM CONTROLLER USING SPRINGFOX
      • STEP 1: ADDING SPRINGFOX SWAGGER DEPENDENCIES INTO OUR PROJECT
      • STEP 2: CREATE A SWAGGER CONFIGURATION CLASS
      • STEP 3: TESTING
    • GENERATE OPENAPI UI FROM CONTROLLER USING OPENAPI
      • STEP 1: ADDING OPENAPI STRINGDOC INTO OUR PROJECT
      • STEP 2: TESTING

    GENERATE SWAGGER UI FROM CONTROLLER USING SPRINGFOX

    STEP 1: ADDING SPRINGFOX SWAGGER DEPENDENCIES INTO OUR PROJECT

    • We will add two spring fox dependencies into file pom.xml of our project as below:
    
     <!--springfox-swagger2 -->
        <dependency>
            <groupId>io.springfox</groupId>
            <artifactId>springfox-swagger2</artifactId>
            <version>2.9.2</version>
        </dependency>


    <!--springfox-swagger-ui -->
    <dependency>
        <groupId>io.springfox</groupId>
        <artifactId>springfox-swagger-ui</artifactId>
        <version>2.9.2</version>
    </dependency>

    STEP 2: CREATE A SWAGGER CONFIGURATION CLASS

    • In this step we will create a class named SwaggerConfiguration and add a simple configuration code as below:
        package com.example.workflow.configurations;

    import org.springframework.context.annotation.Configuration;
    import springfox.documentation.builders.PathSelectors;
    import springfox.documentation.builders.RequestHandlerSelectors;
    import springfox.documentation.spi.DocumentationType;
    import springfox.documentation.spring.web.plugins.Docket;
    import springfox.documentation.swagger2.annotations.EnableSwagger2;

    @Configuration
    @EnableSwagger2
    public class SwaggerConfiguration {

        @Bean
        public Docket docketConfiguration() {
            return new Docket(DocumentationType.SWAGGER_2)
                    .select()
                    .apis(RequestHandlerSelectors.any())
                    .paths(PathSelectors.any())
                    .build();
        }
    }
    • We can seclect which api should be showed or which controller should be show by edit the RequestHandlerSelectors and PathSelectors. For example below, I can choose only apis which is started with /general/** is showed on swagger.
    
    package com.example.workflow.configurations;

    import org.springframework.context.annotation.Bean;
    import org.springframework.context.annotation.Configuration;
    import springfox.documentation.builders.PathSelectors;
    import springfox.documentation.builders.RequestHandlerSelectors;
    import springfox.documentation.spi.DocumentationType;
    import springfox.documentation.spring.web.plugins.Docket;
    import springfox.documentation.swagger2.annotations.EnableSwagger2;

    @Configuration
    @EnableSwagger2
    public class SwaggerConfiguration {

        @Bean
        public Docket docketConfiguration() {
            return new Docket(DocumentationType.SWAGGER_2)
                    .select()
                    .apis(RequestHandlerSelectors.any())
                    .paths(PathSelectors.ant("/general/**"))
                    .build();
        }
    }
    • So this config will load all your Controller with all rest apis that are contained in your project. You can select which controller should be add into the swagger ui by edit the RequestHandlerSelectors and PathSelectors.

    STEP 3: TESTING

    • Let’s create your Controller class which contains some REST apis. In this example I will show all apis in this class on swagger ui. For example:
    
    package com.example.workflow.controller;

    import com.example.workflow.model.NewTable;
    import com.example.workflow.model.gen.CountryNameResponse;
    import com.example.workflow.service.GeneralService;
    import org.springframework.beans.factory.annotation.Autowired;
    import org.springframework.http.MediaType;
    import org.springframework.http.ResponseEntity;
    import org.springframework.web.bind.annotation.*;

    @RestController
    public class GeneralController {


        @Autowired
        private GeneralService generalService;


        @RequestMapping(method = RequestMethod.POST, path = "/primary/newtable", consumes = MediaType.APPLICATION_JSON_VALUE)
        public ResponseEntity<?> saveDataToCamundaNewTable(@RequestBody NewTable newTable) {
            return ResponseEntity.ok(generalService.saveDataToCamundaNewTable(newTable));
        }

        @RequestMapping(method = RequestMethod.POST, path = "/secondary/newtable", consumes = MediaType.APPLICATION_JSON_VALUE)
        public ResponseEntity<?> saveDataToMyDBNewTable(@RequestBody NewTable newTable) {
            return ResponseEntity.ok(generalService.saveDataToMyDB(newTable));
        }

        @RequestMapping(method = RequestMethod.POST, path = "/general/newtable", consumes = MediaType.APPLICATION_JSON_VALUE)
        public ResponseEntity<?> saveDataToAllDBNewTable(@RequestBody NewTable newTable) {
            return ResponseEntity.ok(generalService.saveDataToAllDB(newTable));
        }

        @RequestMapping(method = RequestMethod.GET, path = "/general/history/tasks", produces = MediaType.APPLICATION_JSON_VALUE)
        public ResponseEntity<?> getAllHistoryTasks() {
            return ResponseEntity.ok(generalService.getHistoryTasks());
        }

        @RequestMapping(method = RequestMethod.GET, path = "/general/history/process/key/{processKey}/variables",
                produces = MediaType.APPLICATION_JSON_VALUE)
        public ResponseEntity<?> getAllHistoryVariablesByProcessKey(@PathVariable("processKey") String processKey) {
            return ResponseEntity.ok(generalService.getHistoryVariables(processKey));
        }

        @RequestMapping(method = RequestMethod.GET, path = "/general/soap/country/{countryISO}",
                produces = MediaType.TEXT_XML_VALUE)
        public ResponseEntity<?> getCountryNameByISO(@PathVariable("countryISO") String countryISO) {
            return ResponseEntity.ok(generalService.getCountryName(countryISO));
        }

        @RequestMapping(method = RequestMethod.GET, path = "/general/soap/country/{countryISO}/v2",
                produces = MediaType.TEXT_XML_VALUE)
        public ResponseEntity<CountryNameResponse> getCountryNameByISOV2(@PathVariable("countryISO") String countryISO) {
            return ResponseEntity.ok(generalService.getCountryNameV2(countryISO));
        }
    }
    • So now let run your spring boot service and go to the url localhost:8080/swagger-ui.html then you can see the swagger ui as below

    GENERATE OPENAPI UI FROM CONTROLLER USING OPENAPI

    STEP 1: ADDING OPENAPI STRINGDOC INTO OUR PROJECT

    • We will add two spring fox dependencies into file pom.xml of our project as below:
    
    <!--openapi springdoc -->
    <dependency>
        <groupId>org.springdoc</groupId>
        <artifactId>springdoc-openapi-ui</artifactId>
        <version>1.5.10</version>
    </dependency>
